**思路：**

1. 设定左右边界下标，循环寻找，条件：`while(left < right)`   
2. 算出mid下标，注意溢出：`mid = left + (right - left)/2`   
3. 每次就对比mid对应的数和target的值是否相等

---

## 左闭右开版本
每次不取右边界，则右边界下标就是直接取`nums.size()`

```cpp
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int left = 0;
        int right = nums.size(); // 注意这里可以不减去1，因为不取右边界

        // 左闭右开
        while(left < right){
            int mid = left + (right - left)/2;

            if(nums[mid] < target){
                left = mid + 1;
            }
            else if(target < nums[mid]){
                right = mid;
            }
            else{
                return mid;
            }
        }

        return -1;
    }
};
```

## 左闭右闭版本
取右边界，每次循环后要注意边界移动+1-1

```cpp
class Solution {
public:
    int search(vector<int>& nums, int target) {
        int right = nums.size() - 1; // 注意右边界取值
        int left = 0;
        
        // 左闭右闭
        while(left <= right){
            int mid = left + (right - left)/2;
            
            if(nums[mid] < target){
                left = mid + 1;
            }
            else if(target < nums[mid]){
                right = mid - 1;
            }
            else{
                return mid;
            }
        }

        return -1;
    }
};
```
