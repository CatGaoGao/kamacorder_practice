# 二叉树的层序遍历

力扣链接：https://leetcode.cn/problems/binary-tree-level-order-traversal/

---

**思路：**

1. 用队列先进先出   
2. 一层一层的装进一个数组   
3. while循环中套for循环，每次for循环结束后就结束一层，就把该层装入总的层集合中

---

## 代码

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> result; // 结果集合
        queue<TreeNode*> que; // 遍历元素，先进先出

        if(root == NULL) return result;
        que.push(root);

        while(!que.empty()){
            int n = que.size(); // 留下队列长度
            vector<int> ans; // 用来装一层的元素

            for(int i = 0 ; i < n ; i++){
                TreeNode* cur = que.front(); // 依次取队首元素
                que.pop(); // 出队列
                ans.push_back(cur->val); // 加入到层数组中
                
                if(cur->left) que.push(cur->left); // 判断左右孩子是否为空
                if(cur->right) que.push(cur->right); // 并把它们加入队列等候处理
            }

            result.push_back(ans); // 把一层的结果装进result
        }

        return result;
    }
};
```
